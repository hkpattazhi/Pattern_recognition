def con_matrix(y_true, y_pred):
    """Function to create confusion matrix
        Args:
            y_true: correct classes for the testing points
            y_pred: predicted values for testing points"""
    class_value = list(set(y_true))
    true_positive = 0
    true_negative = 0
    false_positive = 0
    false_negative = 0
    for j in range(len(y_true)):
        if y_true[j] == max(class_value):
            if y_pred[j] == y_true[j]:
                true_negative+=1
            else:
                false_negative+=1
        if y_true[j] == min(class_value):
            if y_pred[j] == y_true[j]:
                true_positive+=1
            else:
                false_positive+=1
    confusion_matrix = [[true_positive, false_positive],
                       [false_negative, true_negative]]
    return confusion_matrix

def knn_cv(train1, train2, test1, test2):
    """Function to find the accuracies for cross validation tables using knn, returns the accuracies 
       and confusion matrix
        Args:
            train1: list of cross validation tables for class1
            train2: list of cross validation tables for class2
            test1: testing points cross validation tables for class1
            test2: testing points cross validation tables for class2"""
    total_accuracy = []
    metrics = {}
    metrics['precision'] = []
    metrics['recall'] = []
    metrics['f1_score'] = []
    confusion_matrix = []
    for i in range(len(train1)):
        nn = knn.KNN(train1[i], train2[i], 3)
        result_class1 = nn.knn_classification(test1[i])
        result_class2 = nn.knn_classification(test2[i])    
        count1 = 0
        count2 = 0
        for j in result_class1:
            if int(j) == 0:
                count1 = count1 +1
        a1 = count1/len(result_class1)
        for j in result_class2:
            if int(j) == 1:
                count2 = count2 +1
        a2 = count2/len(result_class2)
        total_accuracy.append(((a1+a2)/2)*100)
        r = list(result_class1) + list(result_class2)
        y = [0]*len(result_class1) + [1]*len(result_class2)
        metrics['precision'].append(precision_score(y, r, average = 'weighted'))
        metrics['recall'].append(recall_score(y,r, average = 'weighted'))
        metrics['f1_score'].append(f1_score(y, r, average = 'weighted'))
        confusion_matrix.append(con_matrix(y,r))
    return total_accuracy, metrics, confusion_matrix

def predict_cv_ml(train1, train2, test1, test2):
    """Function to find the accuracies and confusion matrix for cross validation tables
       using quadratic classifier: maximum likelihood estimate
        """
    total_accuracy = []
    metrics['precision'] = []
    metrics['recall'] = []
    metrics['f1_score'] = []
    confusion_matrix = []
    
    for i in range(len(train1)):
        r1 = []
        r2 = []
        ml_est_mat1 = ml_estimate.mlEstimate(train1[i])
        mean1 = ml_est_mat1.mean_estimate()
        cov1 = ml_est_mat1.cov_estimate()
        ml_est_mat2 = ml_estimate.mlEstimate(train2[i])
        mean2 = ml_est_mat2.mean_estimate()
        cov2 = ml_est_mat2.cov_estimate()
        A, B, C = bayes_optimal.fit_model(np.array(cov1), np.array(cov2), np.array(mean1), np.array(mean2))
        result1 = bayes_optimal.predict(np.array(test1[i]), A, B, C)
        count1 = 0
        k = result1.shape[1]
        for j in range(len(result1)):
            r1.append(int(result1[j][k-1]))
            if int(result1[j][k-1]) == 0:
                count1 = count1 +1
        a1 = count1/len(result1)
        #print(a1)
        result2 = bayes_optimal.predict(np.array(test2[i]), A, B, C)
        count2 = 0
        for j in range(len(result2)):
            r2.append(int(result2[j][k-1]))
            if int(result2[j][k-1]) == 1:
                count2 = count2+1
        a2 = count2/len(result2)
        #print(a2)
        total_accuracy.append(((a1+a2)/2)*100)
        r = list(r1) + list(r2)
        y = [0]*len(r1) + [1]*len(r2)
        metrics['precision'].append(precision_score(y, r, average = 'weighted'))
        metrics['recall'].append(recall_score(y,r, average = 'weighted'))
        metrics['f1_score'].append(f1_score(y, r, average = 'weighted'))  
        confusion_matrix.append(con_matrix(r,y))
    return total_accuracy, metrics, confusion_matrix

             abs_mean2: absolute mean for class2

             abs_cov2: absolute covariance for class2
             """
    
    total_accuracy = []
    metrics['precision'] = []
    metrics['recall'] = []
    metrics['f1_score'] = []

        cov2 = abs_cov2

def fld_cv(train1, train2, test1, test2):
    """Function to find the accuracy and confusion matrix for cross validation tables 
        using fishers linear discriminant
        Args:
             train1: training list of tables for class1
             train2: training list of tables for class2
             test1: testing points for class1
             test2: testing points for class2"""
    
    total_accuracy = []
    
    metrics['precision']
    
    metrics['recall']
    
    metrics['f1_score']
    
    confusion_matrix = []
    
    for i in range(len(train1)):
        flda = fishers_lda.FishersLDA(train1[i], train2[i])
        result_class1 = flda.predict(test1[i])
        result_class2 = flda.predict(test2[i])
        count1 = 0
        count2 = 0
        for j in result_class1:
            if int(j) == 0:
                count1 = count1 +1
        a1 = count1/len(result_class1)
        for j in result_class2:
            if int(j) == 1:
                count2 = count2 +1
        a2 = count2/len(result_class2)
        total_accuracy.append(((a1+a2)/2)*100)
        
        r = list(result_class1) + list(result_class2)
        
        y = [0]*len(result_class1) + [1]*len(result_class2)
        
        metrics['precision'].append(precision_score(y, r, average = 'weighted'))
        
        metrics['recall'].append(recall_score(y,r, average = 'weighted'))
        
        metrics['f1_score'].append(f1_score(y, r, average = 'weighted'))       
       
        confusion_matrix.append(con_matrix(y,r))
    
    return total_accuracy, metrics, confusion_matrix


def func_cal(x, y, A, B):
    """Calculates the line equation
        Args:
            x: data points for x axis
            y: data points for the y axis
            A: coefficient1
            B: coefficient2
    """
    return (A*x + B*y)

def predict_cv_percep(train1, train2, test1, test2):
    total_accuracy = []
    confusion_matrix = []
    for i in range(len(train1)):
        shape = train1[i].shape[1]
        percep = perceptron.perceptron(shape)
        X, Y = percep.combine_classes(train1[i], train2[i])
        percep.train(X, Y)
        result_class1 = percep.predict(test1[i])
        result_class2 = percep.predict(test2[i])
        count1 = 0
        count2 = 0
        for j in result_class1:
            if int(j) == 0:
                count1 = count1 +1
        a1 = count1/len(result_class1)
        print("a1: "+ str(a1))
        for j in result_class2:
            if int(j) == 1:
                count2 = count2 +1
        a2 = count2/len(result_class2)
        print("a2: "+ str(a2))
        total_accuracy.append(((a1+a2)/2)*100)
        r = list(result_class1) + list(result_class2)
        y = [0]*len(result_class1) + [1]*len(result_class2)
        metrics['precision'].append(precision_score(y, r, average = 'weighted'))
        metrics['recall'].append(recall_score(y,r, average = 'weighted'))
        metrics['f1_score'].append(f1_score(y, r, average = 'weighted'))
        confusion_matrix.append(con_matrix(r, y))
    return total_accuracy, metrics, confusion_matrix
